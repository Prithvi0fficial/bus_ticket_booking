<script>
$(document).ready(function () {
    let selectedSeats = [];
    let bookedSeats = [];
    let seatPrice = {{ seat_price|default:0|floatformat:2 }};
    let seatLayout = JSON.parse('{{ seat_layout|safe }}');

    function fetchBookedSeats() {
        $.ajax({
            url: `/get_booked_seats/{{ bus.id }}/`, // New API endpoint for booked seats
            method: "GET",
            dataType: "json",
            success: function (data) {
                bookedSeats = data.booked_seats; // Update booked seats dynamically
                renderSeats();
            },
            error: function () {
                alert("Error fetching booked seats. Please try again.");
            }
        });
    }

    function renderSeats() {
        let container = $('#seat-container');
        container.empty();
        selectedSeats = [];  // Clear previous selections when reloading

        seatLayout.forEach(row => {
            row.forEach(seat => {
                if (seat === 'aisle') {
                    container.append(`<div class='aisle'></div>`);
                } else if (seat !== '') {
                    let isBooked = bookedSeats.includes(seat);
                    let seatClass = isBooked ? 'seat booked' : 'seat';

                    let seatDiv = `<div class='${seatClass}' data-seat='${seat}'>${seat}</div>`;

                    container.append(seatDiv);
                }
            });
        });

        updateTotal();
    }

    $(document).on("click", ".seat", function () {
        if ($(this).hasClass("booked")) return;

        let seatNumber = $(this).data("seat");

        if ($(this).hasClass("selected")) {
            $(this).removeClass("selected");
            selectedSeats = selectedSeats.filter(s => s !== seatNumber);
        } else {
            $(this).addClass("selected");
            selectedSeats.push(seatNumber);
        }

        updateTotal();
    });

    function updateTotal() {
        let totalSeats = selectedSeats.length;
        let totalPrice = totalSeats * seatPrice;

        $("#total-seats").text(totalSeats);
        $("#total-price").text(totalPrice);
    }

    $("#book-button").click(function () {
        if (selectedSeats.length === 0) {
            alert("Please select at least one seat.");
            return;
        }

        $("#selected-seats").val(selectedSeats.join(","));
        $("#selected-seats-confirmation").val(selectedSeats.join(","));
        $("#total-price-confirmation").val(selectedSeats.length * seatPrice);

        let confirmationUrl = `/confirmation/{{ bus.id }}/?seats=${selectedSeats.join(",")}&price=${selectedSeats.length * seatPrice}`;
        window.location.href = confirmationUrl;
    });

    fetchBookedSeats(); // Get booked seats dynamically
});


</script>
